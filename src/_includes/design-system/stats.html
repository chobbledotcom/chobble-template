{%- comment -%} Stats display as an accessible description list. Parameters (via block object): - block.items: Array of stat objects with 'value' and 'label' properties - block.reveal: If true, adds data-reveal to each item (default: true) Usage: {% assign myStats = "99%|Uptime,50k|Users,24/7|Support" | split: "," %} {% include "design-system/stats.html", block: { items: myStats } %} Or with objects: {% include "design-system/stats.html", block: { items: page.stats } %} (where page.stats = [{value: "99%", label: "Uptime"}, ...]) {%- endcomment -%} {%- assign reveal = block.reveal | default: true -%}
{%- for item in block.items -%} {%- comment -%} Handle both "value|label" strings and {value, label} objects {%- endcomment -%} {%- if item.value -%} {%- assign value = item.value -%} {%- assign label = item.label -%} {%- else -%} {%- assign parts = item | split: "|" -%} {%- assign value = parts[0] -%} {%- assign label = parts[1] -%} {%- endif -%}
{{ label }}
{{ value }} {{ label }}
{%- endfor -%}
